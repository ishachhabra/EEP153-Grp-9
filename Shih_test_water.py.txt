def test_freshwater_withdrawal_non_negative():
    # Query total freshwater withdrawal for the world
    water = freshwater_withdrawal(2010, "WLD")
    # Water withdrawal should never be negative
    assert water >= 0, "Freshwater withdrawal must be non-negative"

def test_water_stress_percentage_range():
    # Retrieve water stress value for a country and year
    stress = water_stress(2015, "USA")
    assert 0 <= stress <= 100, "Water stress should be between 0 and 100 percent"

# Test multiple countries to ensure dataset is not country-specific
def test_water_data_multiple_countries():
    for country in ["USA", "IND", "CHN"]:
        assert freshwater_withdrawal(2010, country) > 0
